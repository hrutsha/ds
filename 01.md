```
from scipy import stats

def detect_outlier_robust(df, col, threshold=3.5):
    median = np.median(df[col])
    mad = stats.median_abs_deviation(df[col])
    
    rz = (df[col] - median) / mad
    
    outliers = df[np.abs(rz) > threshold]
    
    return outliers

# apply
robust_outliers = detect_outlier_robust(df, "Income")
print(robust_outliers)
```
